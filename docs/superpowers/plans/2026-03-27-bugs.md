    req: { path, originalUrl: path },

    next: jest.fn(),

  setupComplete = false;

test('blocks API routes when setup incomplete', () => {
  const { req, res, next } = mockReqRes('/api/library');
  setupMiddleware(req, res, next);
  expect(res.status).toHaveBeenCalledWith(403);

- [ ] **Step 1: Fix the dependency array**

### Task 8: Fix mobile seek not checking cast state (B12)

**Files:**
- Modify: `packages/client/src/components/PlayerBar.jsx:41-42`

**Context:** Desktop seek (line 109) checks `!cast?.isCasting` before seeking local audio. Mobile seek (line 42) doesn't — it always sets `audioRef.current.currentTime`, even when casting. This desyncs cast and local playback.

- [ ] **Step 1: Fix mobile seek handler**

In `packages/client/src/components/PlayerBar.jsx` lines 41-42, change:
```javascript
onClick={(e) => { const rect = e.currentTarget.getBoundingClientRect(); const pctClick = (e.clientX - rect.left) / rect.width;
  if (audioRef.current && duration) audioRef.current.currentTime = pctClick * duration; }}>
```
to:
```javascript
onClick={(e) => { const rect = e.currentTarget.getBoundingClientRect(); const pctClick = (e.clientX - rect.left) / rect.width;
  if (duration) {
    const seekTo = pctClick * duration;
    if (cast?.isCasting && cast.castSeek) { cast.castSeek(Math.round(seekTo)); }
    else if (audioRef.current) { audioRef.current.currentTime = seekTo; }
  } }}>
```
